//******************************************************************************
//メインメニューの展開
//
//AYATEMPLATE.MenuItemの使い方
//1つめの文字列＝選択肢に表示する文字列
//2つめの文字列＝選んだときに実行される関数名
//3つめの文字列＝改行する量　50%刻み　省略すると100%（＝\n）
//******************************************************************************

//------------------------------------------------------------------------------
//メインメニュー展開
//------------------------------------------------------------------------------
OpenMenu
{
	"\0\s[0]メニューなのです。\n\n"
	--
	メニュー("なんか話して","Menu_AITALK") + '\n'
	--
	メニュー("しゃべり頻度を変えて","Menu_TALKINTERVAL") + '\n'
	--
	メニュー("他のゴーストと話す","Menu_GHOSTCOMSTART") + '\n'
	--
	メニュー("AIについて","Menu_ABOUTAYA") + '\n[150]'
	--
	メニュー("なんでもない","Menu_CANCEL")
}

//------------------------------------------------------------------------------
//メニュー選択タイムアウト
//------------------------------------------------------------------------------
選択肢時間切れ
{
	"\0\s[1]‥‥‥\w8し、司令官さん？\e"
}


//******************************************************************************
//各項目の処理内容
//******************************************************************************

//---- 何か話して
Menu_AITALK
{
	//---- 明示的にAIトーク開始を指示
	ランダムトーク実行
}

//---- さっきの話は
Menu_LASTTALK
{
	AYATEMPLATE.LastTalk
}

//---- トーク頻度の調整
Menu_TALKINTERVAL
{
	"\0\s[0]どのくらいがいいのかな？\n\n"
	--
	メニュー("うるさいくらい","Menu_TALKMAX") + '\n'
	--
	メニュー("ふつう","Menu_TALKNORMAL") + '\n'
	--
	メニュー("静か","Menu_TALKPOOR") + '\n'
	--
	メニュー("黙ってて","Menu_TALKNONE") + '\n[150]'
	--
	メニュー("戻る","OpenMenu")
}

//---- 文について説明
Menu_ABOUTAYA
{
	"\0\s[0]ぽにゃづまはYAYAで動作しているのです。ゴーストはややめさんをパクって作ってるのです。\e"
}

//---- 他のゴーストと話す
Menu_GHOSTCOMSTART
{
	StartCommunicate
}

//---- キャンセルされた
Menu_CANCEL
{
	"\0\s[0]キャンセルされたのです。\e"
}

//---- トーク頻度
Menu_TALKMAX
{
	//---- うるさく話す(15秒間隔)
	aitalkinterval = 15
	"\0\s[0]たくさん話すのです。\e"
}

Menu_TALKNORMAL
{
	//---- ふつうに話す(3分間隔)
	aitalkinterval = 180
	"\0\s[0]ふつうくらいに話すのです‥‥ふつうって？\e"
}

Menu_TALKPOOR
{
	//---- 静かに話す(5分間隔)
	aitalkinterval = 300
	"\0\s[0]あまり話さないのです。\e"
}

Menu_TALKNONE
{
	//---- 話さない
	aitalkinterval = 0
	"\0\s[1]‥‥‥‥\e"
}

//---- ヘッドライン
Menu_HEADLINECANCEL
{
	//---- キャンセルされた
	"\0\s[0]キャンセルされたのです。\e"
}
