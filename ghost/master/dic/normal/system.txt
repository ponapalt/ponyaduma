//------------------------------------------------------------------------------
//PCがスリープに入る
//------------------------------------------------------------------------------
OnSysSuspend
{
	<<"
	ぽ０６．２０１５：…眠くなってきたのです…
	れ１０：PCがスリープ入るで。
	ぽ０６：おやすみなのです、（ユーザー）…
	改行無しぽ０４：…ちゃんと起こしてくれるのです…よね…？
	">>
}

//------------------------------------------------------------------------------
//PCがスリープから復帰した
//------------------------------------------------------------------------------
OnSysResume
{
	if reference[0] == "critical"
	{
		<<'
		ぽ０２．２０３２：…はっ！？バッテリーが限界で叩き起こされたのです…！
		れ１１：緊急起動やな。はよ充電せな。
		'>>
	}
	elseif reference[0] == "auto"
	{
		<<'
		ぽ０６：…
		改行無しぽ０２．２０３１：…あれ？勝手に起きたのです？
		れ１０．２０３８：自動復帰やな。何かのスケジュールちゃう。
		ぽ０４：（ユーザー）さんいないのに…ちょっと怖いのです。
		'>>
	}
	else
	{
		<<'
		ぽ０６：…
		改行無しぽ０２：…はっ。
		れ１０．２０３８：おかえり。どんくらい寝てた。
		ぽ０４．２０３５：…ぽにゃも一緒に落ちてたのです…？
		れ１０．２０２１：PCと一蓮托生やからな。
		'>>
	}
}
