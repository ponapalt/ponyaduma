//------------------------------------------------------------------------------
//インストール完了メッセージ
//------------------------------------------------------------------------------
OnInstallComplete
{
	_ref0 = SPLIT(reference[0],' with',2)[0]

	"\0\s[5]"
	--
	if _ref0 == "shell" {
		"シェル"
	}
	elseif _ref0 == "ghost" {
		"ゴースト"
	}
	elseif _ref0 == "balloon" {
		"バルーン"
	}
	elseif _ref0 == "plugin" {
		"プラグイン"
	}
	--
	"のインストール、完了なのです！\n"
	--
	if basewarenameex == 'SSP' {
		_id = ''
		if _ref0 == "shell" {
			_id = 'shell'
		}
		elseif _ref0 == "ghost" {
			_id = 'ghost'
		}
		elseif _ref0 == "balloon" {
			_id = 'balloon'
		}

		if _id != '' {
			"切り替えますか？\n\n"
			--
			メニュー("切り替える","OnAYATEMPLATE.InstallChange",_id) + '\n'
			--
			メニュー("切り替えない","Menu_CANCEL")
		}
	}
}

OnAYATEMPLATE.InstallChange
{
	"\![change,%(reference[0]),lastinstalled]"
}

//------------------------------------------------------------------------------
//インストール失敗メッセージ
//------------------------------------------------------------------------------
OnInstallFailure
{
	if reference[0] == "unlha32"
	{
		<<'
		ぽ０４：箱を開ける道具が見つからないのです…展開用のDLLがないみたいなのです。
		れ１０：工具がないと開封もでけへんな。
		'>>
	}
	elseif reference[0] == "extraction"
	{
		<<'
		ぽ０４．２０３５：中身がぐしゃぐしゃで取り出せないのです…
		れ１０：配送事故やな。
		ぽ０４：もう一度ダウンロードしてみてほしいのです…
		'>>
	}
	elseif reference[0] == "invalid type"
	{
		<<'
		ぽ０４：箱の中に説明書が入ってないのです…install.txtに問題があるみたいなのです。
		れ１０：中身はあるのに開けられへんのか。もどかしいな。
		'>>
	}
	elseif reference[0] == "artificial"
	{
		<<'
		ぽ０２：…あっ、止めたのです？
		れ１０：気が変わったんやろ。
		ぽ０１：ぽにゃ、まだ開けてる途中だったのです…
		'>>
	}
	elseif reference[0] == "unsupported"
	{
		<<'
		ぽ０４：この箱の開け方がわからないのです…対応してない形式みたいなのです。
		れ１０：読めへん言語の箱渡されてもな。
		'>>
	}
	elseif reference[0] == "password"
	{
		<<'
		ぽ０７．２０１８：鍵がかかってて開かないのです…！
		れ１０：パスワードちゃうんやろ。
		ぽ０４：合言葉が違ったのです…確認してほしいのです。
		'>>
	}
	else
	{
		<<'
		ぽ０４：インストールに失敗しちゃったのです…
		れ１０：何がアカンかったんやろな。
		'>>
	}
}
