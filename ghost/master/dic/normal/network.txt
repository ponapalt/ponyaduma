//******************************************************************************

//******************************************************************************

//------------------------------------------------------------------------------
//メールチェック開始
//------------------------------------------------------------------------------
OnBIFFBegin
{
	<<"
	ぽ００：メールをチェックするのです。
	">>
}

//------------------------------------------------------------------------------
//メールチェック成功
//------------------------------------------------------------------------------
OnBIFFComplete
{
	if reference[0] == 0
	{
		<<'
		ぽ００：メールは来てなかったのです。
		れ１０：平和やな。
		'>>
	}
	elseif reference[0] == 1
	{
		<<'
		ぽ０５．２０２３：1通メールが届いてるのです！
		れ１０：誰からやろな。
		'>>
	}
	else
	{
		<<"
		ぽ０５．２０２３：%(reference[0])通もメールが届いてるのです！
		れ１０．２０２１：人気者やな。
		">>
	}
}

//------------------------------------------------------------------------------
//メールチェック失敗
//------------------------------------------------------------------------------
OnBIFFFailure
{
	if reference[0] == "timeout"
	{
		<<'
		ぽ０４：メールサーバーさんが黙り込んでるのです…タイムアウトしちゃったのです。
		れ１０：居留守やな。
		'>>
	}
	elseif reference[0] == "kick"
	{
		<<'
		ぽ０７．２０１８：門前払いされちゃったのです…！
		れ１０：ユーザー名かパスワードちゃうんやろ。
		ぽ０４：合言葉を確認してほしいのです…
		'>>
	}
	elseif reference[0] == "defect"
	{
		<<'
		ぽ０４：そもそも繋がらないのです…
		れ１０：道が通ってへんな。
		ぽ０４：接続先の設定を確認してほしいのです…
		'>>
	}
	else
	{
		<<'
		ぽ０４：メールチェックできなかったのです…
		れ１０：何やろな。
		'>>
	}
}


//******************************************************************************
//ネットワーク更新
//******************************************************************************

//------------------------------------------------------------------------------
//ネットワーク更新開始
//------------------------------------------------------------------------------
OnUpdateBegin
{
	<<"
	ぽ００：ネットワーク更新を始めるのです。新しい自分になれるかもしれないのです。
	">>
}

//------------------------------------------------------------------------------
//ネットワーク更新準備完了
//------------------------------------------------------------------------------
OnUpdateReady
{
	<<"
	ぽ００：%(TOINT(reference[0])+1)個のファイルをダウンロードするのです。
	">>
}

//------------------------------------------------------------------------------
//ネットワーク更新終了
//------------------------------------------------------------------------------
OnUpdateComplete
{
	if reference[0] == "none"
	{
		"\0\s[0]更新はなかったのです。今のままでも十分ってことなのです。\e"
	}
	else
	{
		"\0\s[5]更新完了なのです！ちょっと新しくなったのです！\e"
	}
}

//------------------------------------------------------------------------------
//ネットワーク更新失敗
//------------------------------------------------------------------------------
OnUpdateFailure
{
	if reference[0] == "timeout"
	{
		<<'
		ぽ０４：サーバーさんがお返事してくれないのです…タイムアウトしちゃったのです。
		れ１０：向こうさんお留守やな。
		'>>
	}
	elseif reference[0] == "md5 miss"
	{
		<<'
		ぽ０４．２０３５：届いたファイルの中身が合わないのです…途中で壊れちゃったかもなのです。
		れ１０：検品不合格やな。
		'>>
	}
	elseif reference[0] == "artificial"
	{
		<<'
		ぽ０２：…止めたのです？
		れ１０：途中下車か。
		ぽ０１：ぽにゃ、更新がんばってたのです…
		'>>
	}
	elseif reference[0] == "fileio"
	{
		<<'
		ぽ０４．２０１３：保存する場所が足りないのです…！
		れ１０：ディスクがパンパンやな。
		ぽ０４：ぽにゃの居場所がないのです…！
		'>>
	}
	elseif reference[0] == "readonly"
	{
		<<'
		ぽ０４：ファイルに書き込めないのです…読み取り専用になってるみたいなのです。
		れ１０：見るだけで触れへんやつやな。
		'>>
	}
	elseif reference[0] == "virusdetect"
	{
		<<'
		ぽ０７．２０３６：セキュリティソフトに止められたのです…！
		れ１０：門番に怪しまれたんやな。
		ぽ０７：ぽにゃは怪しくないのです…！
		'>>
	}
	elseif reference[0] == "toomanyredirect"
	{
		<<'
		ぽ０４．２０１４：あっちこっちたらい回しにされて辿り着けないのです…
		れ１０：リダイレクトし過ぎやな。
		'>>
	}
	elseif reference[0] == "executing"
	{
		<<'
		ぽ０４：もうひとつの更新がまだ走ってるのです…
		れ１０：順番待ちやな。
		'>>
	}
	elseif reference[0] == "paramerror"
	{
		<<'
		ぽ０４：更新の設定がおかしいのです…ネットワーク更新のタグを確認してほしいのです。
		れ１０：住所の書き間違いみたいなもんやな。
		'>>
	}
	else
	{
		<<'
		ぽ０４：更新に失敗しちゃったのです…
		れ１０：何が原因やろな。

//******************************************************************************
//ヘッドラインセンス
//******************************************************************************

//------------------------------------------------------------------------------
//ヘッドラインセンス開始
//------------------------------------------------------------------------------
OnHeadlinesenseBegin
{
	<<"
	ぽ００：%(reference[0])のヘッドラインを取得中なのです。
	">>
}

//------------------------------------------------------------------------------
//ヘッドライン読み上げ
//------------------------------------------------------------------------------
OnHeadlinesense.OnFind
{
	if reference[2] == "First" || reference[2] == "First and Last"
	{
		"\0\b2\s[0]%(reference[0])のヘッドラインなのです。"
	}
	elseif reference[2] == "Next"
	{
		"\0\b2\s[0]%(reference[0])ヘッドライン、続きなのです。"
	}
	elseif reference[2] == "Last"
	{
		"\0\b2\s[0]%(reference[0])ヘッドライン、最後のページなのです。"
	}
	--
	"\n\n%(reference[3])\n\n"
	--
	if reference[2] == "First" || reference[2] == "Next"
	{
		メニュー("次のページ","") + '\n'
		--
		メニュー("サイトへ",reference[1]) + '\n[150]'
		--
		メニュー("やめ","Menu_HEADLINECANCEL")
	}
	else
	{
		メニュー("サイトへ",reference[1]) + '\n[150]'
		--
		メニュー("やめ","Menu_HEADLINECANCEL")
	}
}

//------------------------------------------------------------------------------
//ヘッドライン更新が無かった
//------------------------------------------------------------------------------
OnHeadlinesenseComplete
{
	<<"
	ぽ００：%(reference[0])に更新はなかったのです。
	">>
}

//------------------------------------------------------------------------------
//ヘッドラインセンス失敗
//------------------------------------------------------------------------------
OnHeadlinesenseFailure
{
	if reference[0] == "can't download"
	{
		<<'
		ぽ０４：ヘッドラインを取りに行ったのに、何も持って帰れなかったのです…
		れ１０：手ぶらで帰還か。
		'>>
	}
	elseif reference[0] == "can't analyze"
	{
		<<'
		ぽ０４：ヘッドラインは届いたのに、読めないのです…
		れ１０：解読不能やな。暗号か。
		ぽ０８：…ぽにゃの読解力の問題じゃないのです。
		'>>
	}
	else
	{
		<<'
		ぽ０４：ヘッドラインの取得に失敗しちゃったのです…
		れ１０：何がアカンかったんやろな。
		'>>
	}
}


//******************************************************************************
//時計合わせ
//******************************************************************************

//------------------------------------------------------------------------------
//時計合わせ開始
//------------------------------------------------------------------------------
OnSNTPBegin
{
	<<"
	ぽ００：時計を合わせるのです。時間のバランスも大事なのです。
	">>
}

//------------------------------------------------------------------------------
//時刻取得成功、比較開始
//------------------------------------------------------------------------------

SNTPCompare_StrForm
{
	_txt = SPLIT(_argv[0],',')
	STRFORM("$04d/$02d/$02d $02d:$02d:$02d",TOINT(_txt[0]),TOINT(_txt[1]),TOINT(_txt[2]),TOINT(_txt[3]),TOINT(_txt[4]),TOINT(_txt[5]))
}

OnSNTPCompare
{
	"\0\s[0]\_q/
	修正前 %(SNTPCompare_StrForm(reference[2]))\n/
	修正後 %(SNTPCompare_StrForm(reference[1]))\n/
	\_q\n"
	--
	if reference[3] == 0
	{
		"ズレてなかったのです。時間にも正確、さすがなのです。\e"
	}
	else
	{
		//\6は補正実行タグ。消しちゃだめ
		"\6\0\n\s[0]%(reference[3])秒ズレてたのです。直しておいたのです。\e"
	}
}

//------------------------------------------------------------------------------
//時計合わせ失敗
//------------------------------------------------------------------------------
OnSNTPFailure
{
	<<"
	ぽ０４：%(reference[0])さんに聞きに行ったのに、お返事もらえなかったのです…
	れ１０：時報屋さんお休みか。
	ぽ０４：時計が合わせられなかったのです…
	">>
}

	：ぽにゃの思考みたいにもたついてるのです…
	れ１０：自覚あったんか。

	">>

	ぽ０２：違うのです！更新チェックなのです！

	">>
}

//------------------------------------------------------------------------------
//ネットワーク接続状態が変化した
//------------------------------------------------------------------------------
OnNetworkStatusChange
{
	// ステータスが変わってなければ何もしない
	if NetworkStatus == reference[0]
	{
		return
	}
	NetworkStatus = reference[0]

	// NOTIFYリクエストではスクリプトを返さない
	if REQ.COMMAND == 'NOTIFY'
	{
		return
	}

	if reference[0] == "online"
	{
		<<"
		ぽ０５．２０４３：ネットワークが繋がったのです！
		れ１０：復旧したな。
		">>
	}
	else
	{
		<<"
		ぽ０４．２０１９：ネットワークが切れたのです…！
		れ１０：オフラインやな。
		ぽ０４．２０１４：ぽにゃ、孤立してるのです…
		改行無しぽ０６：…逆に集中できる…？
		れ１０．２０２１：没頭の口実にすな。
		">>
	}
}
