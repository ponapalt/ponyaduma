//------------------------------------------------------------------------------
//インストール完了メッセージ
//------------------------------------------------------------------------------
OnInstallComplete
{
	_ref0 = SPLIT(reference[0],' with',2)[0] //hogehoge with hemohemo

	"\0\s[0]"
	--
	if _ref0 == "shell" {
		"シェル"
	}
	elseif _ref0 == "ghost" {
		"ゴースト"
	}
	elseif _ref0 == "balloon" {
		"バルーン"
	}
	elseif _ref0 == "plugin" {
		"プラグイン"
	}
	--
	"インストール完了なのです。\n"
	--
	if basewarenameex == 'SSP' {
		_id = ''
		if _ref0 == "shell" {
			_id = 'shell'
		}
		elseif _ref0 == "ghost" {
			_id = 'ghost'
		}
		elseif _ref0 == "balloon" {
			_id = 'balloon'
		}

		if _id != '' {
			"切り替えますか？\n\n"
			--
			メニュー("切り替える","OnAYATEMPLATE.InstallChange",_id) + '\n'
			--
			メニュー("切り替えない","Menu_CANCEL")
		}
	}
}

//------------------------------------------------------------------------------
//メールチェック開始
//------------------------------------------------------------------------------
OnBIFFBegin
{
	<<"
	ぽ０：メールが届いてないかチェックするのです。
	">>
}

//------------------------------------------------------------------------------
//メールチェック成功
//------------------------------------------------------------------------------
OnBIFFComplete
{
	if reference[0] == 0
	{
		//---- メールは無かった
		"\0\s[0]メールは無かったのです。\e"
	}
	else
	{
		//---- メールが届いていた
		"\0\s[0]%(reference[0])通のメールがあったのです。\e"
	}
}

//------------------------------------------------------------------------------
//メールチェック失敗
//------------------------------------------------------------------------------
OnBIFFFailure
{
	if reference[0] == "timeout"
	{
		//---- タイムアウト
		"\0\s[1]メールサーバからの応答がないのです…\e"
	}
	elseif reference[0] == "kick"
	{
		//---- 蹴られた
		"\0\s[1]蹴られたのです…\w4ユーザー名かパスワードが間違っていませんか？\e"
	}
	elseif reference[0] == "defect"
	{
		//---- 設定誤り
		"\0\s[1]接続できないのです。設定を確認してほしいのです。\e"
	}
	else
	{
		//---- 原因不明
		"\0\s[1]よく分からないけどメールチェックできないのです…\e"
	}
}

//ネットワーク更新終了
//------------------------------------------------------------------------------

{
	if reference[0] == "none"
	{

		"\0\s[0]更新は無かったのです。\e"
	}
	else
	{
		//---- 実際に更新が完了した
		"\0\s[0]更新完了したのです。\e"
	}
}

//------------------------------------------------------------------------------
//ネットワーク更新失敗
//------------------------------------------------------------------------------
OnUpdateFailure
{
	if reference[0] == "timeout"
	{
		//---- タイムアウトした
		"\0\s[1]タイムアウトなのです…"
	}
	elseif reference[0] == "md5 miss"
	{
		//---- MD5値が一致しなかった
		"\0\s[1]ハッシュが一致しないのです…"
	}
	else
	{
		//---- 上記以外の理由で失敗した
		"\0\s[1]%(reference[0]) で失敗なのです…"
	}
	--
	"\e"
}

//ネットワーク更新 ダウンロード開始

//------------------------------------------------------------------------------
//ヘッドラインセンス開始
//------------------------------------------------------------------------------
OnHeadlinesenseBegin
{
	<<"
	ぽ０：%(reference[0])ヘッドライン取得中。
	">>
}

//------------------------------------------------------------------------------
//ヘッドライン読み上げ
//------------------------------------------------------------------------------
OnHeadlinesense.OnFind
{
	//---- はじめのセリフ
	if reference[2] == "First" || reference[2] == "First and Last"
	{
		//---- 最初のページ
		"\0\b2\s[0]%(reference[0])のヘッドラインなのです。"
	}
	elseif reference[2] == "Next"
	{
		//---- 真ん中あたりのページ
		"\0\b2\s[0]%(reference[0])ヘッドライン、続きなのです。"
	}
	elseif reference[2] == "Last"
	{
		//---- 最後のページ
		"\0\b2\s[0]%(reference[0])ヘッドライン、最後のページなのです。"
	}
	--
	//---- ヘッドライン本体
	"\n\n%(reference[3])\n\n"
	--
	//---- 選択肢
	if reference[2] == "First" || reference[2] == "Next"
	{
		//---- 次のページがある場合
		メニュー("次のページ","") + '\n'
		--
		メニュー("サイトへ",reference[1]) + '\n[150]'
		--
		メニュー("やめ","Menu_HEADLINECANCEL")
	}
	else
	{
		//---- 次のページがない場合
		メニュー("サイトへ",reference[1]) + '\n[150]'
		--
		メニュー("やめ","Menu_HEADLINECANCEL")
	}
}

//------------------------------------------------------------------------------
//ヘッドライン更新が無かった
//------------------------------------------------------------------------------
OnHeadlinesenseComplete
{
	<<"
	ぽ０：%(reference[0])、更新はないみたいです。
	">>
}

//------------------------------------------------------------------------------
//ヘッドラインセンス失敗
//------------------------------------------------------------------------------
OnHeadlinesenseFailure
{
	"\0\s[0]ヘッドラインセンスに失敗したのです。\n\n理由："
	--
	if reference[0] == "can't download"
	{
		"ダウンロード失敗。"
	}
	elseif reference[0] == "can't analyze"
	{
		"ヘッドライン解析失敗。"
	}
	else
	{
		"謎。"
	}
	--
	"\e"
}

//******************************************************************************
//時計合わせ
//みか (http://homepage2.nifty.com/ko-ki/) の辞書を参考にしました。感謝＞こーき氏
//******************************************************************************

//------------------------------------------------------------------------------
//時計合わせ開始
//------------------------------------------------------------------------------
OnSNTPBegin
{
	<<"
	ぽ０：時刻同期するのです。
	">>
}

//------------------------------------------------------------------------------
//時刻取得成功、比較開始
//この関数で 2008/03/25 19:15:05 のように見やすい形に作り変えます。
//詳しくはSTRFORM関数の説明を見てね！
//わからなければこのままで。
//------------------------------------------------------------------------------

SNTPCompare_StrForm
{
	_txt = SPLIT(_argv[0],',')
	STRFORM("$04d/$02d/$02d $02d:$02d:$02d",TOINT(_txt[0]),TOINT(_txt[1]),TOINT(_txt[2]),TOINT(_txt[3]),TOINT(_txt[4]),TOINT(_txt[5]))
}

OnSNTPCompare
{
	"\0\s[0]\_q/
	修正前 %(SNTPCompare_StrForm(reference[2]))\n/
	修正後 %(SNTPCompare_StrForm(reference[1]))\n/
	\_q\n"
	--
	if reference[3] == 0
	{
		"特にズレていないみたいなのです。\e"
	}
	else
	{
		//\6は補正実行タグ。消しちゃだめ
		"\6\0\n\s[0]%(reference[3])秒ズレていたのです。\e"
	}
}

//------------------------------------------------------------------------------
//時計合わせ失敗
//------------------------------------------------------------------------------
OnSNTPFailure
{
	<<"
	ぽ１：時刻同期に失敗したのです…
	">>
}
