//------------------------------------------------------------------------------
//インストール完了メッセージ
//------------------------------------------------------------------------------
OnInstallComplete
{
	_ref0 = SPLIT(reference[0],' with',2)[0]

	"\0\s[5]"
	--
	if _ref0 == "shell" {
		"シェル"
	}
	elseif _ref0 == "ghost" {
		"ゴースト"
	}
	elseif _ref0 == "balloon" {
		"バルーン"
	}
	elseif _ref0 == "plugin" {
		"プラグイン"
	}
	--
	"のインストール、完了なのです！\n"
	--
	if basewarenameex == 'SSP' {
		_id = ''
		if _ref0 == "shell" {
			_id = 'shell'
		}
		elseif _ref0 == "ghost" {
			_id = 'ghost'
		}
		elseif _ref0 == "balloon" {
			_id = 'balloon'
		}

		if _id != '' {
			"切り替えますか？\n\n"
			--
			メニュー("切り替える","OnAYATEMPLATE.InstallChange",_id) + '\n'
			--
			メニュー("切り替えない","Menu_CANCEL")
		}
	}
}

OnAYATEMPLATE.InstallChange
{
	"\![change,%(reference[0]),lastinstalled]"
}

//------------------------------------------------------------------------------

OnBIFFBegin
{
	<<"
	ぽ００：メールをチェックするのです。
	">>
}

//------------------------------------------------------------------------------
//メールチェック成功
//------------------------------------------------------------------------------
OnBIFFComplete
{
	if reference[0] == 0
	{
		"\0\s[0]メールは来てなかったのです。\e"
	}
	else
	{
		"\0\s[5]%(reference[0])通のメールが届いてるのです！\e"
	}
}

//------------------------------------------------------------------------------
//メールチェック失敗
//------------------------------------------------------------------------------
OnBIFFFailure
{
	if reference[0] == "timeout"
	{
		"\0\s[4]メールサーバから応答がないのです……\e"
	}
	elseif reference[0] == "kick"
	{
		"\0\s[4]接続を拒否されちゃったのです……ユーザー名かパスワードを確認してほしいのです。\e"
	}
	elseif reference[0] == "defect"
	{
		"\0\s[4]接続できないのです。設定を確認してほしいのです。\e"
	}
	else
	{
		"\0\s[4]メールチェックできなかったのです……原因がわからないのです。\e"
	}
}

//------------------------------------------------------------------------------
//時計合わせ開始
//------------------------------------------------------------------------------
OnSNTPBegin
{
	<<"
	ぽ００：時計を合わせるのです。時間のバランスも大事なのです。
	">>
}

//------------------------------------------------------------------------------
//時刻取得成功、比較開始
//------------------------------------------------------------------------------

SNTPCompare_StrForm
{
	_txt = SPLIT(_argv[0],',')
	STRFORM("$04d/$02d/$02d $02d:$02d:$02d",TOINT(_txt[0]),TOINT(_txt[1]),TOINT(_txt[2]),TOINT(_txt[3]),TOINT(_txt[4]),TOINT(_txt[5]))
}

OnSNTPCompare
{
	"\0\s[0]\_q/
	修正前 %(SNTPCompare_StrForm(reference[2]))\n/
	修正後 %(SNTPCompare_StrForm(reference[1]))\n/
	\_q\n"
	--
	if reference[3] == 0
	{
		"ズレてなかったのです。時間にも正確、さすがなのです。\e"
	}
	else
	{
		//\6は補正実行タグ。消しちゃだめ
		"\6\0\n\s[0]%(reference[3])秒ズレてたのです。直しておいたのです。\e"
	}
}

//------------------------------------------------------------------------------
//時計合わせ失敗
//------------------------------------------------------------------------------
OnSNTPFailure
{
	<<"
	ぽ０４：時刻の同期に失敗しちゃったのです……
	">>
}
