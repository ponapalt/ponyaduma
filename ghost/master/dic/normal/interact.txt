//******************************************************************************

//------------------------------------------------------------------------------
//着せ替えが変更された
//------------------------------------------------------------------------------
OnDressupChanged
{
	if reference[2] == 1
	{
		if reference[1] == "水ばぐゅう"
		{
			<<"
			ぽ０２．２０３８：わっ、涼しそうなのです…！
			れ１０．２０２１：夏仕様やな。
			ぽ０５．２０３３：水も滴るいいばぐゅうなのです！
			れ．２０５０：誰がうまいこと言えと。
			">>
		}
		elseif reference[1] == "青ばぐゅう"
		{
			<<"
			ぽ０５．２０２４：おっ、青いのです！
			れ１０：落ち着いた感じやろ。
			ぽ０３．２０３７：…ぽにゃの目の色とお揃いなのです？
			れ．２０２１：そこ気にするんや。
			">>
		}
		elseif reference[1] == "内燃ばぐゅう"
		{
			<<"
			ぽ０２．２０３１：え…？
			改行無しぽ０４：なんで機関車になってるのです…？
			れ１１．２０２１：DE10やな。渋いチョイスやで。
			ぽ０６：…
			改行無しぽ０５．２０４３：…嫌いじゃないのです。
			">>
		}
		else
		{
			<<"
			ぽ０３．２０３７：…あっ、着けてくれたのです？
			れ１０：似合っとるで。
			ぽ０５．２０３３：えへへ、ありがとうなのです！
			">>
		}
	}
	else
	{
		<<"
		ぽ００：あっ、いつものばぐゅうに戻ったのです。
		れ１０：灰色も悪くないで。
		ぽ０５：落ち着くのです。
		">>
	}
}
